    setSelectedBoxIds((current) => {
      if (current.includes(boxId)) {
        return current.filter((id) => id !== boxId);
      }
      return [...current, boxId];
    });
  }

    setSelectedBoxIds(DEFAULT_SELECTED_BOX_IDS);
